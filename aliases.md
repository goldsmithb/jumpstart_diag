# Copy the following lines to your .zshrc file and either relaunch terminal or run
# `source ~/.zshrc` to start using the aliases

alias ..-"cd .."

# Git-related Aliases
# s = git status
alias s="git status"
# gaa = git add all
alias gaa="git add ."
# c <message> = git commit -m <message>
alias c="git commit -m"
